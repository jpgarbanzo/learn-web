1. Variables and Constants

JavaScript provides three ways to declare variables:

// Variable declarations let userName = "Alice"; // Can be reassigned const PI = 3.14159; // Cannot be reassigned var oldStyle = "avoid this"; // Function-scoped (avoid in modern JS) // Reassigning variables userName = "Bob"; // This works with let // PI = 3.14; // This would cause an error! console.log("User name:", userName); console.log("PI constant:", PI);

2. Data Types

Primitive Data Types

// String - text data let message = "Hello, World!"; let singleQuotes = 'Also a string'; let templateLiteral = `Template literal with ${message}`; // Number - integers and decimals let age = 25; let price = 19.99; let negative = -42; // Boolean - true or false let isActive = true; let isComplete = false; // Undefined - variable declared but not assigned let undefinedVar; // Null - intentional absence of value let emptyValue = null; console.log("String:", typeof message); console.log("Number:", typeof age); console.log("Boolean:", typeof isActive); console.log("Undefined:", typeof undefinedVar); console.log("Null:", typeof emptyValue); // Note: this shows "object" (JavaScript quirk)

Reference Types

// Arrays - ordered lists let fruits = ["apple", "banana", "orange"]; let numbers = [1, 2, 3, 4, 5]; let mixed = [1, "hello", true, null]; // Objects - key-value pairs let person = { name: "John Doe", age: 30, city: "New York", isEmployed: true }; // Accessing array elements console.log("First fruit:", fruits[0]); console.log("Array length:", fruits.length); // Accessing object properties console.log("Person name:", person.name); console.log("Person age:", person["age"]); // Alternative syntax

3. JavaScript Operators

Arithmetic Operators

let a = 10; let b = 3; console.log("Addition:", a + b); // 13 console.log("Subtraction:", a - b); // 7 console.log("Multiplication:", a * b); // 30 console.log("Division:", a / b); // 3.3333... console.log("Modulo (remainder):", a % b); // 1 console.log("Exponentiation:", a ** b); // 1000 // String concatenation with + let firstName = "John"; let lastName = "Doe"; let fullName = firstName + " " + lastName; console.log("Full name:", fullName);

Comparison Operators

let x = 5; let y = "5"; // Loose equality (with type coercion) console.log("x == y:", x == y); // true (5 == "5") console.log("x == '5':", x == "5"); // true // Strict equality (no type coercion) console.log("x === y:", x === y); // false (5 !== "5") console.log("x === 5:", x === 5); // true // Other comparison operators console.log("x > 3:", x > 3); // true console.log("x < 10:", x < 10); // true console.log("x >= 5:", x >= 5); // true console.log("x <= 4:", x <= 4); // false

Logical Operators

let isLoggedIn = true; let hasPermission = false; let isAdmin = true; // AND operator (&&) - both conditions must be true console.log("Logged in AND has permission:", isLoggedIn && hasPermission); // false console.log("Logged in AND is admin:", isLoggedIn && isAdmin); // true // OR operator (||) - at least one condition must be true console.log("Has permission OR is admin:", hasPermission || isAdmin); // true console.log("Has permission OR logged in:", hasPermission || isLoggedIn); // true // NOT operator (!) - opposite of the condition console.log("NOT logged in:", !isLoggedIn); // false console.log("NOT has permission:", !hasPermission); // true // Complex logical expressions let canAccess = isLoggedIn && (hasPermission || isAdmin); console.log("Can access:", canAccess); // true

4. Template Literals

let name = "Alice"; let age = 28; let city = "Boston"; // Old way - string concatenation let oldMessage = "Hello, my name is " + name + " and I am " + age + " years old."; // New way - template literals let newMessage = `Hello, my name is ${name} and I am ${age} years old.`; // Multi-line strings let address = ` Name: ${name} Age: ${age} City: ${city} Country: USA `; // Expressions in template literals let calculation = `The sum of 5 and 3 is ${5 + 3}.`; let comparison = `Is ${age} greater than 18? ${age > 18 ? 'Yes' : 'No'}`; console.log("Old message:", oldMessage); console.log("New message:", newMessage); console.log("Address:", address); console.log("Calculation:", calculation); console.log("Comparison:", comparison);

5. Practical Examples

Working with Arrays

let shoppingCart = []; let itemPrices = [12.99, 8.50, 23.75, 5.25]; // Adding items to cart shoppingCart.push("T-shirt"); shoppingCart.push("Book"); shoppingCart.push("Headphones"); console.log("Shopping cart:", shoppingCart); console.log("Cart length:", shoppingCart.length); // Calculate total price let total = 0; for (let i = 0; i < itemPrices.length; i++) { total += itemPrices[i]; } console.log(`Total price: $${total.toFixed(2)}`);

Working with Objects

let student = { firstName: "Sarah", lastName: "Johnson", age: 20, courses: ["Math", "Physics", "Computer Science"], isEnrolled: true, // Method (function inside object) getFullName: function() { return this.firstName + " " + this.lastName; } }; // Accessing properties console.log("Student name:", student.getFullName()); console.log("Courses:", student.courses.join(", ")); // Adding new properties student.gpa = 3.8; student.graduationYear = 2024; console.log("Updated student:", student);
